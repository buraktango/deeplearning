\section*{Exercise \chapnumber.1}

\mA

\begin{bmatrix}
    1 & 3 & 5 \\
    2 & 4 & 6 \\

\begin{equation}
\mB

\begin{bmatrix}
    7 & 2 \\

\begin{enumerate}[label=(\alph*)]

\item $\mA\mB$
\item $\mA\odot\mA$
