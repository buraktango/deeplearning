\section*{Exercise \chapnumber.1}
Consider the following matrices,

\begin{equation}
\mA
=
\begin{bmatrix}
    1 & 3 & 5 \\
    2 & 4 & 6 \\
    0 & 8 & 2 
\end{bmatrix}
\end{equation}

\begin{equation}
\mB
=
\begin{bmatrix}
    7 & 2 \\
    1 & 5 \\
    9 & 4 
\end{bmatrix}
\end{equation}

Calculate the following values/matrices:

\begin{enumerate}[label=(\alph*)]

\item $\textit{A}_{2,3}$
\item $\mA^T$
\item $\mB^T$
\item $\mA+\mA$
\item $2\mB+1$
\item $\mA\mA$
\item $\mA\mB$
\item $\mA\odot\mA$
\item $(\mI_{3}\mB)\mI_{2}$

\end{enumerate}
