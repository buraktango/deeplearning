\mA

\begin{bmatrix}
    1 & 3 & 5 \\
    2 & 4 & 6 \\

\begin{equation}
\mB
=
\begin{bmatrix}
    7 & 2 \\
    1 & 5 \\
    9 & 4

\item $\mA\mA$
\item $\mA\mB$
\item $\mA\odot\mA$
\item $(\mI_{3}\mB)\mI_{2}$

\end{enumerate}
